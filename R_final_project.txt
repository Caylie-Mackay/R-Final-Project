library(tidyverse)
library(ggridges)

df <- read_csv("f1_strategy_dataset_v4.csv")

df

df %>% 
  select(Year) %>% 
  distinct()

df %>% 
  group_by(Year) %>% 
  count()

#This Dataset has data from 4 different years, 2022 with 21860 rows, 2023 with 24914 rows, 2024 with 27557 rows, 2025 with 27040 rows.

df %>% 
  select(Compound) %>% 
  distinct()

# This dataset showcases 5 different tire compounds. MEDIUM, HARD, SOFT, INTERMEDIATE, WET.

df %>% 
  group_by(Compound) %>% 
  summarise(avg_lap_time = mean(`LapTime (s)`, na.rm=T))

#This code shows us the average lap time across all drivers for each tire compound.

df %>% 
  filter(Position <= 10) %>% 
  group_by(Compound) %>% 
  summarise(avg_tire_deg = mean(Cumulative_Degradation, na.ra=T))

#This code shows the average cumulative tire degridation by compound of datapoints taken from racers in the top 10 positions.

max(df$RaceProgress)

min(df$RaceProgress)

#This dataset has data points from 1.3% race progress to 98.7% race progress.

df %>% 
  filter(PitStop == 1) %>% 
  summarise(avg_positions_lost <- mean(LapTime_Delta, na.ra=T))

#This code shows us the averave change in lap time from the previous lap for drivers who made a pit stop.


# Using a 2 tailed T-Test we will test to see if a driver makes a pit stop is related to the lap number.
#Null Hypothesis: If a driver makes a pit stop it does not effect what lap number they pitted on.
#Alternative Hypothesis: If a driver makes a pit stop it will effect what lap number they pitted on.

as.character(df$PitStop)

alpha = 0.05

results <- t.test(LapNumber ~ PitStop, data=df)

p <- results$p.value

if(p <= alpha){
  print("Rejected")
} else {
  print("Failed to reject")
}

#The null hypothesis was rejected so that means that there is a relationship bettween a driver making a pit stop and the lap number they pitted at.
